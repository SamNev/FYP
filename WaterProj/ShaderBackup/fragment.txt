#version 130

uniform float u_MaxHeight;
uniform vec3 u_Color;
uniform bool u_IgnoreHeight;

in VS_OUT {
    vec3 FragPos;
} fs_in;

void main()
{
    if(!u_IgnoreHeight)
    {
        vec3 col = fs_in.FragPos.y > 0.0 ? u_Color * (0.2 + (fs_in.FragPos.y/u_MaxHeight) * 0.8) : vec3(0.0, 0.2, -(fs_in.FragPos.y - 0.2f));
        gl_FragColor = vec4(col, 1.0);
    }
    else
    {
	gl_FragColor = vec4(u_Color, 1.0);
    }
}
